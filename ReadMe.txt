PWD Shortcodes

----------------------------------------------------

Container

	[container][/container]

	Optional:

		class 
			Default = ''
			Adds a class to the div

----------------------------------------------------

Row

	[row][/row]

----------------------------------------------------

Col

	[col][/col]

	Required:

		cols
			Default = ''
			Adds column width e.g., "eight"

	Optional:

		offset
			Default = ''
			Adds offset to column e.g., "one"

		class
			Default = ''
			Adds class to columns section
----------------------------------------------------

Link

	[link]

	Required:

		href
			Default = ''
			adds location for link to go to

		content
			Default = ''
			adds the link text

	Optional:

		class
			Default =''
			adds class(es)

----------------------------------------------------
